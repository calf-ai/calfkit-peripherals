1"""Configurable tool-output truncation limits. 

2 

3Ported from anomalyco/opencode PR #23770 (``feat(truncate): allow 

4configuring tool output truncation limits``). 

5 

6OpenCode hardcoded ``MAX_LINES = 2000`` and ``MAX_BYTES = 50 * 1024`` 

7as tool-output truncation thresholds. Hermes-agent had the same 

8hardcoded constants in two places: 

9 

10* ``tools/terminal_tool.py`` — ``MAX_OUTPUT_CHARS = 50000`` (terminal 

11 stdout/stderr cap) 

12* ``tools/file_operations.py`` — ``MAX_LINES = 2000`` / 

13 ``MAX_LINE_LENGTH = 2000`` (read_file pagination cap + per-line cap) 

14 

15This module centralises those values behind a single config section 

16(``tool_output`` in ``config.yaml``) so power users can tune them 

17without patching the source. The existing hardcoded numbers remain as 

18defaults, so behaviour is unchanged when the config key is absent. 

19 

20Example ``config.yaml``:: 

21 

22 tool_output: 

23 max_bytes: 100000 # terminal output cap (chars) 

24 max_lines: 5000 # read_file pagination + truncation cap 

25 max_line_length: 2000 # per-line length cap before '... [truncated]' 

26 

27The limits reader is defensive: any error (missing config file, invalid 

28value type, etc.) falls back to the built-in defaults so tools never 

29fail because of a malformed config. 

30""" 

31 

32from __future__ import annotations 

33 

34from typing import Any, Dict 

35 

36# Hardcoded defaults — these match the pre-existing values, so adding 

37# this module is behaviour-preserving for users who don't set 

38# ``tool_output`` in config.yaml. 

39DEFAULT_MAX_BYTES = 50_000 # terminal_tool.MAX_OUTPUT_CHARS 

40DEFAULT_MAX_LINES = 2000 # file_operations.MAX_LINES 

41DEFAULT_MAX_LINE_LENGTH = 2000 # file_operations.MAX_LINE_LENGTH 

42 

43# Module-level cache — populated on first call. 

44# Avoids repeated config file I/O on every tool call. 

45_cached_limits: dict | None = None 

46 

47 

48def _coerce_positive_int(value: Any, default: int) -> int: 

49 """Return ``value`` as a positive int, or ``default`` on any issue.""" 

50 try: 

51 iv = int(value) 

52 except (TypeError, ValueError): 

53 return default 

54 if iv <= 0: 

55 return default 

56 return iv 

57

59def get_tool_output_limits() -> Dict[str, int]: 

60 """Return resolved tool-output limits, reading ``tool_output`` from config. 

61 

62 Keys: ``max_bytes``, ``max_lines``, ``max_line_length``. Missing or 

63 invalid entries fall through to the ``DEFAULT_*`` constants. This 

64 function NEVER raises. 

65 

66 Result is cached for the process lifetime to avoid repeated disk I/O 

67 on every tool call. Call ``_reset_tool_output_limits_cache()`` in 

68 tests that need a fresh read after config changes. 

69 """ 

70 global _cached_limits 

71 if _cached_limits is not None: 

72 return _cached_limits 

73 try: 

74 from calfkit_tools.hermes._shims.hermes_cli.config import load_config 

75 cfg = load_config() or {} 

76 section = cfg.get("tool_output") if isinstance(cfg, dict) else None 

77 if not isinstance(section, dict): 

78 section = {} 

79 except Exception: 

80 section = {} 

81 

82 _cached_limits = { 

83 "max_bytes": _coerce_positive_int(section.get("max_bytes"), DEFAULT_MAX_BYTES), 

84 "max_lines": _coerce_positive_int(section.get("max_lines"), DEFAULT_MAX_LINES), 

85 "max_line_length": _coerce_positive_int( 

86 section.get("max_line_length"), DEFAULT_MAX_LINE_LENGTH 

87 ), 

88 } 

89 return _cached_limits 

90 

91 

92def _reset_tool_output_limits_cache() -> None: 

93 """Reset the cached limits — for tests or after config hot-reload.""" 

94 global _cached_limits 

95 _cached_limits = None 

96 

97 

98def get_max_bytes() -> int: 

99 """Shortcut for terminal-tool callers that only need the byte cap.""" 

100 return get_tool_output_limits()["max_bytes"] 

101 

102 

103def get_max_lines() -> int: 

104 """Shortcut for file-ops callers that only need the line cap."""

106 

107 

108def get_max_line_length() -> int: 

109 """Shortcut for file-ops callers that only need the per-line cap.""" 

110 return get_tool_output_limits()["max_line_length"]
