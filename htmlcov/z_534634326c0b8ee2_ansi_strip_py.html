1"""Strip ANSI escape sequences from subprocess output. 

2 

3Used by terminal_tool, code_execution_tool, and process_registry to clean 

4command output before returning it to the model. This prevents ANSI codes 

5from entering the model's context — which is the root cause of models 

6copying escape sequences into file writes. 

7 

8Covers the full ECMA-48 spec: CSI (including private-mode ``?`` prefix, 

9colon-separated params, intermediate bytes), OSC (BEL and ST terminators), 

10DCS/SOS/PM/APC string sequences, nF multi-byte escapes, Fp/Fe/Fs 

11single-byte escapes, and 8-bit C1 control characters. 

12""" 

13 

14import re 

15 

16_ANSI_ESCAPE_RE = re.compile( 

17 r"\x1b" 

18 r"(?:" 

19 r"\[[\x30-\x3f]*[\x20-\x2f]*[\x40-\x7e]" # CSI sequence 

20 r"|\][\s\S]*?(?:\x07|\x1b\\)" # OSC (BEL or ST terminator) 

21 r"|[PX^_][\s\S]*?(?:\x1b\\)" # DCS/SOS/PM/APC strings 

22 r"|[\x20-\x2f]+[\x30-\x7e]" # nF escape sequences 

23 r"|[\x30-\x7e]" # Fp/Fe/Fs single-byte 

24 r")" 

25 r"|\x9b[\x30-\x3f]*[\x20-\x2f]*[\x40-\x7e]" # 8-bit CSI 

26 r"|\x9d[\s\S]*?(?:\x07|\x9c)" # 8-bit OSC 

27 r"|[\x80-\x9f]", # Other 8-bit C1 controls 

28 re.DOTALL, 

29) 

30 

31# Fast-path check — skip full regex when no escape-like bytes are present. 

32_HAS_ESCAPE = re.compile(r"[\x1b\x80-\x9f]") 

33 

34 

35def strip_ansi(text: str) -> str: 

36 """Remove ANSI escape sequences from text. 

37 

38 Returns the input unchanged (fast path) when no ESC or C1 bytes are 

39 present. Safe to call on any string — clean text passes through 

40 with negligible overhead. 

41 """ 

42 if not text or not _HAS_ESCAPE.search(text): 

43 return text 

44 return _ANSI_ESCAPE_RE.sub("", text)
