1"""File tool nodes: read_file, write_file, patch, search_files. 

2 

3Each node is a thin 1:1 wrapper over a vendored hermes file tool, driven 

4wholesale through the ``registry.dispatch`` seam (``node/_runtime.py``) — never 

5through vendor internals. Parameters mirror the upstream JSON schemas the tools 

6registered for the LLM (``file_tools.py``: ``READ_FILE_SCHEMA``, 

7``WRITE_FILE_SCHEMA``, ``PATCH_SCHEMA``, ``SEARCH_FILES_SCHEMA``), minus the 

8injected ``task_id``, with a leading ``ToolContext``. Tenancy (ADR-0004): the 

9per-caller ``session_key`` is fed to hermes as its ``task_id``, so the vendored 

10per-``task_id`` file trackers provide isolation. See docs/design/node-port.md §3. 

11""" 

12

15from typing import Annotated, Literal 

16 

17from calfkit import ToolContext, agent_tool 

18from pydantic import Field

20from ._runtime import dispatch, session_key

23@agent_tool 

24def read_file( 

25 ctx: ToolContext, 

26 path: str, 

27 offset: Annotated[int, Field(ge=1)] = 1, 

28 limit: Annotated[int, Field(le=2000)] = 500, 

29) -> dict: 

30 """Read a text file with line numbers and pagination. 

31 

32 Use this instead of cat/head/tail in terminal. Output format: 

33 'LINE_NUM|CONTENT'. Suggests similar filenames if not found. Use offset and 

34 limit for large files. Reads exceeding ~100K characters are rejected; use 

35 offset and limit to read specific sections of large files. NOTE: Cannot read 

36 images or binary files — use vision_analyze for images. 

37 

38 Args: 

39 path: Path to the file to read (absolute, relative, or ~/path). 

40 offset: Line number to start reading from (1-indexed, default: 1). 

41 limit: Maximum number of lines to read (default: 500, max: 2000). 

42 """ 

43 return dispatch( 

44 "read_file", 

45 {"path": path, "offset": offset, "limit": limit}, 

46 session_key=session_key(ctx), 

47 )

50@agent_tool 

51def write_file(ctx: ToolContext, path: str, content: str, cross_profile: bool = False) -> dict: 

52 """Write content to a file, completely replacing existing content. 

53 

54 Use this instead of echo/cat heredoc in terminal. Creates parent directories 

55 automatically. OVERWRITES the entire file — use 'patch' for targeted edits. 

56 Auto-runs syntax checks on .py/.json/.yaml/.toml and other linted languages; 

57 only NEW errors introduced by this write are surfaced (pre-existing errors 

58 are filtered out). 

59 

60 Args: 

61 path: Path to the file to write (will be created if it doesn't exist, 

62 overwritten if it does). 

63 content: Complete content to write to the file. 

64 cross_profile: Opt out of the cross-profile soft guard. Defaults to 

65 false. Set true ONLY after explicit user direction to edit another 

66 Hermes profile's skills/plugins/cron/memories — by default these 

67 writes are blocked with a warning because they affect a different 

68 profile than the one this session is running under. 

69 """ 

70 return dispatch( 

71 "write_file", 

72 {"path": path, "content": content, "cross_profile": cross_profile}, 

73 session_key=session_key(ctx), 

74 )

77@agent_tool 

78def patch( 

79 ctx: ToolContext, 

80 mode: Literal["replace", "patch"] = "replace", 

81 path: str | None = None, 

82 old_string: str | None = None, 

83 new_string: str | None = None, 

84 replace_all: bool = False, 

85 patch: str | None = None, 

86 cross_profile: bool = False, 

87) -> dict: 

88 """Targeted find-and-replace edits in files. 

89 

90 Use this instead of sed/awk in terminal. Uses fuzzy matching (9 strategies) 

91 so minor whitespace/indentation differences won't break it. Returns a 

92 unified diff. Auto-runs syntax checks after editing. 

93 

94 REPLACE MODE (mode='replace', default): find a unique string and replace it. 

95 REQUIRED PARAMETERS: mode, path, old_string, new_string. 

96 PATCH MODE (mode='patch'): apply V4A multi-file patches for bulk changes. 

97 REQUIRED PARAMETERS: mode, patch. 

98 

99 Args: 

100 mode: Edit mode. 'replace' (default): requires path + old_string + 

101 new_string. 'patch': requires patch content only. 

102 path: REQUIRED when mode='replace'. File path to edit. 

103 old_string: REQUIRED when mode='replace'. Exact text to find and replace. 

104 Must be unique in the file unless replace_all=true. Include 

105 surrounding context lines to ensure uniqueness. 

106 new_string: REQUIRED when mode='replace'. Replacement text. Pass empty 

107 string '' to delete the matched text. 

108 replace_all: Replace all occurrences instead of requiring a unique match 

109 (default: false). 

110 patch: REQUIRED when mode='patch'. V4A format patch content. Format: 

111 *** Begin Patch 

112 *** Update File: path/to/file 

113 @@ context hint @@ 

114 context line 

115 -removed line 

116 +added line 

117 *** End Patch 

118 cross_profile: Opt out of the cross-profile soft guard. Defaults to 

119 false. Set true ONLY after explicit user direction to edit another 

120 Hermes profile's skills/plugins/cron/memories. 

121 """ 

122 return dispatch( 

123 "patch", 

124 { 

125 "mode": mode, 

126 "path": path, 

127 "old_string": old_string, 

128 "new_string": new_string, 

129 "replace_all": replace_all, 

130 "patch": patch, 

131 "cross_profile": cross_profile, 

132 }, 

133 session_key=session_key(ctx), 

134 )

137@agent_tool 

138def search_files( 

139 ctx: ToolContext, 

140 pattern: str, 

141 target: Literal["content", "files"] = "content", 

142 path: str = ".", 

143 file_glob: str | None = None, 

144 limit: int = 50, 

145 offset: int = 0, 

146 output_mode: Literal["content", "files_only", "count"] = "content", 

147 context: int = 0, 

148) -> dict: 

149 """Search file contents or find files by name. 

150 

151 Use this instead of grep/rg/find/ls in terminal. Ripgrep-backed, faster than 

152 shell equivalents. 

153 

154 Content search (target='content'): Regex search inside files. Output modes: 

155 full matches with line numbers, file paths only, or match counts. 

156 

157 File search (target='files'): Find files by glob pattern (e.g., '*.py', 

158 '*config*'). Also use this instead of ls — results sorted by modification 

159 time. 

160 

161 Args: 

162 pattern: Regex pattern for content search, or glob pattern (e.g., '*.py') 

163 for file search. 

164 target: 'content' searches inside file contents, 'files' searches for 

165 files by name. One of: 'content', 'files'. 

166 path: Directory or file to search in (default: current working 

167 directory). 

168 file_glob: Filter files by pattern in grep mode (e.g., '*.py' to only 

169 search Python files). 

170 limit: Maximum number of results to return (default: 50). 

171 offset: Skip first N results for pagination (default: 0). 

172 output_mode: Output format for grep mode: 'content' shows matching lines 

173 with line numbers, 'files_only' lists file paths, 'count' shows match 

174 counts per file. One of: 'content', 'files_only', 'count'. 

175 context: Number of context lines before and after each match (grep mode 

176 only). 

177 """ 

178 return dispatch( 

179 "search_files", 

180 { 

181 "pattern": pattern, 

182 "target": target, 

183 "path": path, 

184 "file_glob": file_glob, 

185 "limit": limit, 

186 "offset": offset, 

187 "output_mode": output_mode, 

188 "context": context, 

189 }, 

190 session_key=session_key(ctx), 

191 )
