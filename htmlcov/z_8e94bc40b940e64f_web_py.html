1"""The ``web_search`` and ``web_extract`` tool nodes. 

2 

3Design: docs/design/node-port.md §3; docs/design/web-search-tool-port.md §4/§7; 

4ADR-0002 (env-var provider selection; the config resolver stays dormant). 

5 

6Both nodes are **stateless** — no ctx, no tenancy. At import the node registers 

7its own name->class provider list (replacing the dropped plugin hooks), each 

8guarded by ``ImportError`` because the backends are extras-dependent. Per call 

9the active backend is chosen from an env var via ``registry.get_provider`` — the 

10vendored config resolver (``get_active_*``) is never called (ADR-0002). 

11 

12- ``web_search`` (sync): ``WEB_SEARCH_BACKEND`` (default ``ddgs``). Returns the 

13 provider's ``search()`` contract dict as-is; clean ``error`` dict on an 

14 unknown/unavailable backend. 

15- ``web_extract`` (async): ``WEB_EXTRACT_BACKEND`` (default ``tavily``). Every 

16 URL is run through the vendored ``async_is_safe_url`` (fail-closed SSRF guard) 

17 before dispatch; unsafe URLs become per-URL error entries and never reach the 

18 provider. Safe URLs go to ``provider.extract`` (sync -> ``asyncio.to_thread``). 

19 Results are re-wrapped into ``{"success": True, "data": [...]}`` (the wrapping 

20 the dropped dispatcher used to do; web-search-tool-port.md §7). 

21 

22Provider exceptions PROPAGATE: neither node wraps ``provider.search`` / 

23``provider.extract`` in a ``try``/``except`` — an unexpected raise bubbles up so 

24calfkit converts it into a ``FailedToolCall``. (The clean ``error`` dicts these 

25nodes return cover only *known* conditions — unknown/unavailable backend, 

26unsupported capability, blocked URL. In practice the shipped providers 

27self-handle network errors and return an error dict of their own.) 

28"""

81@agent_tool 

82def web_search(query: str, limit: int = 5) -> dict: 

83 """Search the web and return ranked links. 

84 

85 Args: 

86 query: The search query. 

87 limit: Maximum number of results to return (default 5). 

88 """ 

89 # Deliberately sync: calfkit runs sync tools in a thread pool, which already 

90 # satisfies the design's "offload the blocking provider call off the event 

91 # loop" requirement — do NOT "fix" this to a bare async def (that would run 

92 # the blocking provider.search ON the event loop, without offload). 

93 name = os.getenv("WEB_SEARCH_BACKEND", DEFAULT_SEARCH_BACKEND) 

94 provider = get_provider(name) 

95 if provider is None: 

96 return {"success": False, "error": f"Unknown web search backend '{name}'"} 

97 if not provider.is_available(): 

98 return { 

99 "success": False, 

100 "error": f"Web search backend '{name}' is not available (missing API key or dependency)", 

101 } 

102 return provider.search(query, limit=limit)

106async def web_extract(urls: list[str]) -> dict: 

107 """Extract readable content from one or more web pages. 

108 

109 Result order is NOT guaranteed to match the input ``urls`` order: the 

110 returned ``data`` list is the provider-extracted (safe-URL) entries followed 

111 by the per-URL SSRF-blocked entries. Correlate results by their ``url`` 

112 field, not by position. 

113 

114 Args: 

115 urls: The page URLs to extract content from. 

116 """

120 return {"success": False, "error": f"Unknown web extract backend '{name}'"} 

121 if not provider.supports_extract(): 

122 return { 

123 "success": False, 

124 "error": f"Web extract backend '{name}' does not support extract",

129 "error": f"Web extract backend '{name}' is not available (missing API key or dependency)",

132 # SSRF guard every URL before dispatch; unsafe URLs become per-URL error 

133 # entries and are never handed to the provider (fail-closed). 

134 safe_urls: list[str] = [] 

135 blocked: list[dict[str, Any]] = [] 

136 for url in urls: 

137 if await async_is_safe_url(url): 

138 safe_urls.append(url) 

139 else: 

140 blocked.append( 

141 { 

142 "url": url, 

143 "title": "", 

144 "content": "", 

145 "error": "URL blocked: private, loopback, or cloud-metadata target", 

146 } 

147 ) 

148 

149 extracted: list[dict[str, Any]] = [] 

150 if safe_urls: 

151 # provider.extract is sync (httpx.post) -> offload off the event loop. 

152 extracted = await asyncio.to_thread(provider.extract, safe_urls) 

153 

154 # Re-wrap into the documented success envelope (web-search-tool-port.md §7); 

155 # the node owns the wrapping the dropped dispatcher used to do. 

156 return {"success": True, "data": [*extracted, *blocked]}
