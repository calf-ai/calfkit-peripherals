1"""Regex-based secret redaction for logs and tool output. 

2 

3Applies pattern matching to mask API keys, tokens, and credentials 

4before they reach log files, verbose output, or gateway logs. 

5 

6Short tokens (< 18 chars) are fully masked. Longer tokens preserve 

7the first 6 and last 4 characters for debuggability.

155_SIGNAL_PHONE_RE = re.compile(r"(\+[1-9]\d{6,14})(?![A-Za-z0-9])")

157# URLs containing query strings — matches `scheme://...?...[# or end]`. 

158# Used to scan text for URLs whose query params may contain secrets. 

159# Ported from nearai/ironclaw#2529.

168# URLs containing userinfo — `scheme://user:password@host` for ANY scheme

196def mask_secret( 

197 value: str, 

198 *, 

199 head: int = 4, 

200 tail: int = 4, 

201 floor: int = 12, 

202 placeholder: str = "***", 

203 empty: str = "", 

204) -> str: 

205 """Mask a secret for display, preserving ``head`` and ``tail`` characters. 

206 

207 Canonical helper for display-time redaction across Hermes — used by 

208 ``hermes config``, ``hermes status``, ``hermes dump``, and anywhere 

209 a secret needs to be shown truncated for debuggability while still 

210 keeping the bulk hidden. 

211 

212 Args: 

213 value: The secret to mask. ``None``/empty returns ``empty``. 

214 head: Leading characters to preserve. Default 4. 

215 tail: Trailing characters to preserve. Default 4. 

216 floor: Values shorter than ``head + tail + floor_margin`` are 

217 fully masked (returns ``placeholder``). Default 12 — 

218 matches the existing config/status/dump convention. 

219 placeholder: Value returned for too-short inputs. Default ``"***"``. 

220 empty: Value returned when ``value`` is falsy (None, ""). The 

221 caller can override this to e.g. ``color("(not set)", 

222 Colors.DIM)`` for user-facing display. 

223 

224 Examples: 

225 >>> mask_secret("sk-proj-abcdef1234567890") 

226 'sk-p...7890' 

227 >>> mask_secret("short") # fully masked 

228 '***' 

229 >>> mask_secret("") # empty default 

230 '' 

231 >>> mask_secret("", empty="(not set)") # empty override 

232 '(not set)' 

233 >>> mask_secret("long-token", head=6, tail=4, floor=18) 

234 '***' 

235 """ 

236 if not value: 

237 return empty 

238 if len(value) < floor: 

239 return placeholder 

240 return f"{value[:head]}...{value[-tail:]}"

243def _mask_token(token: str) -> str: 

244 """Mask a log token — conservative 18-char floor, preserves 6 prefix / 4 suffix.""" 

245 # Empty input: historically this returned "***" rather than "". Preserve. 

246 if not token: 

247 return "***" 

248 return mask_secret(token, head=6, tail=4, floor=18)

273def _redact_url_query_params(text: str) -> str: 

274 """Scan text for URLs with query strings and redact sensitive params. 

275 

276 Catches opaque tokens that don't match vendor prefix regexes, e.g. 

277 `https://example.com/cb?code=ABC123&state=xyz` → `...?code=***&state=xyz`. 

278 """ 

279 def _sub(m: re.Match) -> str: 

280 scheme = m.group(1) 

281 authority = m.group(2) 

282 path = m.group(3) 

283 query = _redact_query_string(m.group(4)) 

284 fragment = m.group(5) or "" 

285 return f"{scheme}://{authority}{path}?{query}{fragment}" 

286 return _URL_WITH_QUERY_RE.sub(_sub, text)

326def redact_sensitive_text(text: str, *, force: bool = False, code_file: bool = False) -> str: 

327 """Apply all redaction patterns to a block of text. 

328 

329 Safe to call on any string -- non-matching text passes through unchanged. 

330 Enabled by default. Disable via security.redact_secrets: false in config.yaml. 

331 Set force=True for safety boundaries that must never return raw secrets 

332 regardless of the user's global logging redaction preference. 

333 

334 Set code_file=True to skip the ENV-assignment and JSON-field regex 

335 patterns when the text is known to be source code (e.g. MAX_TOKENS=*** 

336 constants, "apiKey": "test" fixtures). Prefix patterns, auth headers, 

337 private keys, DB connstrings, JWTs, and URL secrets are still redacted. 

338 

339 Performance: each regex pattern is gated behind a cheap substring 

340 pre-check (e.g. ``"=" in text`` for ENV assignments, ``"://" in text`` 

341 for URLs, ``"eyJ" in text`` for JWTs). On a typical hermes log line 

342 (no secrets) this drops the 13-pattern scan from ~5.6us to ~1.8us per 

343 record (-68%). The pre-checks are conservative — false positives 

344 still run the full regex, which then doesn't match. False negatives 

345 are impossible because every regex requires the gated substring to 

346 match. 

347 """ 

348 if text is None: 

349 return None 

350 if not isinstance(text, str): 

351 text = str(text) 

352 if not text: 

353 return text 

354 if not (force or _REDACT_ENABLED): 

355 return text 

356 

357 # Known prefixes (sk-, ghp_, etc.) — gate on substring presence 

358 if _has_known_prefix_substring(text): 

359 text = _PREFIX_RE.sub(lambda m: _mask_token(m.group(1)), text) 

360 

361 # ENV assignments: OPENAI_API_KEY=*** (skip for code files — false positives) 

362 if not code_file: 

363 if "=" in text: 

364 def _redact_env(m): 

365 name, quote, value = m.group(1), m.group(2), m.group(3) 

366 return f"{name}={quote}{_mask_token(value)}{quote}" 

367 text = _ENV_ASSIGN_RE.sub(_redact_env, text) 

368 

369 # JSON fields: "apiKey": "***" (skip for code files — false positives) 

370 if ":" in text and '"' in text: 

371 def _redact_json(m): 

372 key, value = m.group(1), m.group(2) 

373 return f'{key}: "{_mask_token(value)}"' 

374 text = _JSON_FIELD_RE.sub(_redact_json, text) 

375 

376 # Authorization headers — _AUTH_HEADER_RE is "Authorization: Bearer ..." 

377 # case-insensitive, so "uthorization" is the cheapest substring gate that 

378 # covers both "Authorization" and "authorization" without a casefold(). 

379 if "uthorization" in text or "UTHORIZATION" in text: 

380 text = _AUTH_HEADER_RE.sub( 

381 lambda m: m.group(1) + _mask_token(m.group(2)), 

382 text, 

383 ) 

384 

385 # Telegram bot tokens — pattern requires ":<token>" with digits prefix 

386 if ":" in text: 

387 def _redact_telegram(m): 

388 prefix = m.group(1) or "" 

389 digits = m.group(2) 

390 return f"{prefix}{digits}:***" 

391 text = _TELEGRAM_RE.sub(_redact_telegram, text) 

392 

393 # Private key blocks 

394 if "BEGIN" in text and "-----" in text: 

395 text = _PRIVATE_KEY_RE.sub("[REDACTED PRIVATE KEY]", text) 

396 

397 # Database connection string passwords 

398 if "://" in text: 

399 text = _DB_CONNSTR_RE.sub(lambda m: f"{m.group(1)}***{m.group(3)}", text) 

400 

401 # JWT tokens (eyJ... — base64-encoded JSON headers) 

402 if "eyJ" in text: 

403 text = _JWT_RE.sub(lambda m: _mask_token(m.group(0)), text) 

404 

405 # NOTE: Web-URL redaction (query params + userinfo + HTTP access-log 

406 # request targets) is intentionally OFF. Many legitimate workflows pass 

407 # opaque tokens through query strings — magic-link checkouts, OAuth 

408 # callbacks the agent is meant to follow, pre-signed share URLs — and 

409 # blanket-redacting param values by name breaks those skills mid-flow. 

410 # Known credential shapes (sk-, ghp_, JWTs, etc.) inside URLs are still 

411 # caught by _PREFIX_RE and _JWT_RE above. DB connection-string passwords 

412 # are still caught by _DB_CONNSTR_RE. 

413 

414 # Form-urlencoded bodies (only triggers on clean k=v&k=v inputs). 

415 if "&" in text and "=" in text: 

416 text = _redact_form_body(text) 

417 

418 # E.164 phone numbers (Signal, WhatsApp) 

419 if "+" in text: 

420 def _redact_phone(m): 

421 phone = m.group(1) 

422 if len(phone) <= 8: 

423 return phone[:2] + "****" + phone[-2:] 

424 return phone[:4] + "****" + phone[-4:] 

425 text = _SIGNAL_PHONE_RE.sub(_redact_phone, text) 

426 

427 return text

430# Substrings used to gate ``_PREFIX_RE`` execution. If none of these appear in 

431# the input string, the prefix regex cannot match anything, so we skip it. 

432# False positives are fine (they just run the regex, which then matches 

433# nothing) — the bound is "no false negatives" and that holds because every 

434# pattern in ``_PREFIX_PATTERNS`` has at least one of these as a literal 

435# substring of its leading characters. 

436# 

437# Derived automatically from ``_PREFIX_PATTERNS`` at module load time so a 

438# future PR that adds a new prefix to the regex list can't silently break 

439# the screen.

442 """Return the leading literal characters of a regex pattern. 

443

445 ``?``, ``*``, ``+``, ``|``, ``{``, ``^``, ``$``). Returns the literal 

446 that any match of the pattern MUST contain as a substring, so the 

447 pre-screen never produces false negatives. 

448 """ 

449 meta = "[(\\.?*+|{^$" 

450 for i, ch in enumerate(pattern): 450 ↛ 453line 450 didn't jump to line 453 because the loop on line 450 didn't complete

451 if ch in meta: 

452 return pattern[:i] 

453 return pattern
