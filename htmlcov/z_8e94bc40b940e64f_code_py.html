1"""The ``execute_code`` tool node: a sandboxed Python REPL with an RPC tool bridge. 

2 

3Design: docs/design/node-port.md §3 "execute_code specifics"; ADR-0004 (tenancy). 

4 

5A 1:1 calfkit tool node over the vendored hermes ``execute_code`` tool. The model 

6hands it a Python script; the script runs in a sandboxed child process that can 

7call back into a subset of hermes tools (``from hermes_tools import ...``) over an 

8RPC bridge. The node drives the tool wholesale through the ``dispatch`` seam — no 

9reaching into vendor internals — forwarding two extra kwargs: 

10 

11- ``task_id`` (= ``session_key(ctx)``): the tenancy key. In-script tool calls 

12 inherit this same ``task_id`` (the RPC bridge forwards it), so the sandbox 

13 shares the node session's shell environment and file trackers — isolation 

14 propagates (ADR-0004). 

15- ``enabled_tools``: which tool stubs the sandbox may import. The sandbox uses 

16 the intersection of this set with upstream's ``SANDBOX_ALLOWED_TOOLS`` 

17 (terminal, read_file, write_file, patch, search_files, web_search, 

18 web_extract), so listing a non-sandboxable tool (process, todo) is harmless. 

19 

20Trust model (see NODE.md): a script runs arbitrary Python on the node host — the 

21same trust class as ``terminal``. Deploy sandboxed or single-tenant. 

22 

23Sync by design: the upstream handler blocks (it spawns and waits on the child 

24process); calfkit runs sync tools in its thread pool. 

25""" 

26 

27from __future__ import annotations 

28 

29import os 

30 

31from calfkit import ToolContext, agent_tool 

32 

33from calfkit_tools.hermes.node._runtime import dispatch, session_key 

34 

35# The hermes tool set this node exposes to the sandbox by default: every other 

36# hermes tool minus ``execute_code`` itself (a script can't recursively spawn 

37# the sandbox). Upstream intersects this with SANDBOX_ALLOWED_TOOLS, so the 

38# non-sandboxable members (``process``, ``todo``) simply never generate a stub. 

39_DEFAULT_ENABLED_TOOLS = frozenset( 

40 {"terminal", "process", "read_file", "write_file", "patch", "search_files", "todo"} 

41) 

42 

43_ENABLED_TOOLS_ENV = "EXECUTE_CODE_ENABLED_TOOLS" 

44 

45 

46def _enabled_tools() -> set[str]: 

47 """Resolve the sandbox-exposed tool set from node config. 

48 

49 ``EXECUTE_CODE_ENABLED_TOOLS`` is a comma-separated list of tool names; a 

50 blank/whitespace-only value falls back to the default set. Node config, not 

51 agent-authored — the agent only supplies ``code``. 

52 """ 

53 raw = os.environ.get(_ENABLED_TOOLS_ENV) 

54 if raw is None: 

55 return set(_DEFAULT_ENABLED_TOOLS) 

56 names = {name.strip() for name in raw.split(",") if name.strip()} 

57 return names or set(_DEFAULT_ENABLED_TOOLS) 

58 

59 

60@agent_tool 

61def execute_code(ctx: ToolContext, code: str) -> dict | str: 

62 """Run a Python script that can call Hermes tools programmatically. Use this 

63 when you need 3+ tool calls with processing logic between them, need to 

64 filter/reduce large tool outputs before they enter your context, need 

65 conditional branching (if X then Y else Z), or need to loop (fetch N pages, 

66 process N files, retry on failure). 

67 

68 Use normal tool calls instead when: single tool call with no processing, you 

69 need to see the full result and apply complex reasoning, or the task requires 

70 interactive user input. 

71 

72 Hermes tools are imported with ``from hermes_tools import ...`` (e.g. 

73 terminal, read_file, write_file, patch, search_files). terminal() is 

74 foreground-only (no background or pty). Scripts run in the node's working 

75 directory with the active venv's python; in-script tool calls (terminal, 

76 file tools) operate in your own session's shell environment. 

77 

78 Limits: 5-minute timeout, 50KB stdout cap, max 50 tool calls per script. 

79 Print your final result to stdout. Use Python stdlib (json, re, math, csv, 

80 datetime, collections, etc.) for processing between tool calls. Also built in 

81 (no import): json_parse(text), shell_quote(s), retry(fn, max_attempts, delay). 

82 

83 Args: 

84 code: Python code to execute. Import tools with 

85 ``from hermes_tools import ...`` and print your final result to 

86 stdout. 

87 """ 

88 from calfkit_tools.hermes._vendor.tools.code_execution_tool import SANDBOX_ALLOWED_TOOLS 

89 

90 enabled = _enabled_tools() 

91 # Fail closed: upstream expands an EMPTY intersection with 

92 # SANDBOX_ALLOWED_TOOLS back to the FULL set — a typo'd or 

93 # non-sandboxable-only override would silently grant every tool. 

94 if not (enabled & SANDBOX_ALLOWED_TOOLS): 

95 return { 

96 "error": ( 

97 "EXECUTE_CODE_ENABLED_TOOLS resolves to no sandbox-callable tools " 

98 f"(got: {sorted(enabled)}; sandbox-callable: " 

99 f"{sorted(SANDBOX_ALLOWED_TOOLS)}). Refusing to run rather than " 

100 "fall back to the full tool set." 

101 ) 

102 } 

103 return dispatch( 

104 "execute_code", 

105 {"code": code}, 

106 session_key=session_key(ctx), 

107 # Upstream types this Optional[List[str]] and rebuilds a set internally; 

108 # pass a list to match the documented contract. 

109 enabled_tools=sorted(enabled), 

110 )
