2""" 

3Fuzzy Matching Module for File Operations 

4 

5Implements a multi-strategy matching chain to robustly find and replace text, 

6accommodating variations in whitespace, indentation, and escaping common 

7in LLM-generated code. 

8 

9The 8-strategy chain (inspired by OpenCode), tried in order: 

101. Exact match - Direct string comparison 

112. Line-trimmed - Strip leading/trailing whitespace per line 

123. Whitespace normalized - Collapse multiple spaces/tabs to single space 

134. Indentation flexible - Ignore indentation differences entirely 

145. Escape normalized - Convert \\n literals to actual newlines 

156. Trimmed boundary - Trim first/last line whitespace only 

167. Block anchor - Match first+last lines, use similarity for middle 

178. Context-aware - 50% line similarity threshold 

18 

19Multi-occurrence matching is handled via the replace_all flag. 

20 

21Usage:

23

25 content="def foo():\\n pass", 

26 old_string="def foo():", 

27 new_string="def bar():", 

28 replace_all=False 

29 ) 

30"""

50def fuzzy_find_and_replace(content: str, old_string: str, new_string: str, 

51 replace_all: bool = False) -> Tuple[str, int, Optional[str], Optional[str]]: 

52 """ 

53 Find and replace text using a chain of increasingly fuzzy matching strategies. 

54 

55 Args: 

56 content: The file content to search in 

57 old_string: The text to find 

58 new_string: The replacement text 

59 replace_all: If True, replace all occurrences; if False, require uniqueness 

60 

61 Returns: 

62 Tuple of (new_content, match_count, strategy_name, error_message) 

63 - If successful: (modified_content, number_of_replacements, strategy_used, None) 

64 - If failed: (original_content, 0, None, error_description) 

65 """ 

66 if not old_string: 

67 return content, 0, None, "old_string cannot be empty" 

68 

69 if old_string == new_string: 

70 return content, 0, None, "old_string and new_string are identical" 

71 

72 # Try each matching strategy in order 

73 strategies: List[Tuple[str, Callable]] = [ 

74 ("exact", _strategy_exact), 

75 ("line_trimmed", _strategy_line_trimmed), 

76 ("whitespace_normalized", _strategy_whitespace_normalized), 

77 ("indentation_flexible", _strategy_indentation_flexible), 

78 ("escape_normalized", _strategy_escape_normalized), 

79 ("trimmed_boundary", _strategy_trimmed_boundary), 

80 ("unicode_normalized", _strategy_unicode_normalized), 

81 ("block_anchor", _strategy_block_anchor), 

82 ("context_aware", _strategy_context_aware), 

83 ]

86 matches = strategy_fn(content, old_string) 

87 

88 if matches:

90 if len(matches) > 1 and not replace_all: 

91 return content, 0, None, ( 

92 f"Found {len(matches)} matches for old_string. " 

93 f"Provide more context to make it unique, or use replace_all=True." 

94 ) 

95 

96 # Escape-drift guard: when the matched strategy is NOT `exact`, 

97 # we matched via some form of normalization. If new_string 

98 # contains shell/JSON-style escape sequences (\' or \") that 

99 # would be written literally into the file but the matched 

100 # region of the file has no such sequences, this is almost 

101 # certainly tool-call serialization drift — the model typed 

102 # an apostrophe/quote and the transport added a stray 

103 # backslash. Writing new_string as-is would corrupt the file. 

104 # Block with a helpful error so the model re-reads and retries 

105 # instead of the caller silently persisting garbage (or not). 

106 if strategy_name != "exact": 

107 drift_err = _detect_escape_drift(content, matches, old_string, new_string) 

108 if drift_err: 

109 return content, 0, None, drift_err 

110 

111 # Perform replacement. When the matched strategy is NOT `exact`, 

112 # the file's indentation may differ from what the LLM sent in 

113 # old_string/new_string — e.g. LLM used 2-space indent but the 

114 # file is 4-space. Shift new_string by the indentation delta so 

115 # the replacement matches the file's actual indent pattern. 

116 # LLMs frequently serialize tabs / carriage returns in JSON 

117 # tool-call arguments as the two-character sequences ``\t`` and 

118 # ``\r`` (backslash + letter) instead of the real control bytes. 

119 # If we write new_string verbatim, the file ends up with literal 

120 # backslash sequences where the surrounding code uses real tabs. 

121 # 

122 # Strategy: only unescape when the matched region of the file 

123 # *actually contains* the corresponding real control character. 

124 # That mirrors the region-based heuristic in 

125 # ``_detect_escape_drift`` and keeps legitimate writes of the 

126 # literal two-character string ``"\t"`` (e.g. patching Python 

127 # source that contains a tab string literal in source text) 

128 # untouched — those files have a backslash+t in the matched 

129 # region, not a real tab, so we leave new_string alone. 

130 # 

131 # ``\n`` is intentionally excluded: newlines serialize correctly 

132 # through JSON, and rewriting backslash-n would mangle escape 

133 # sequences in source code constants far more often than help. 

134 effective_new = _maybe_unescape_new_string( 

135 new_string, content, matches, 

136 ) 

137 new_content = _apply_replacements( 

138 content, matches, effective_new, 

139 old_string=old_string if strategy_name != "exact" else None, 

140 ) 

141 return new_content, len(matches), strategy_name, None 

142 

143 # No strategy found a match 

144 return content, 0, None, "Could not find a match for old_string in the file"

147def _detect_escape_drift(content: str, matches: List[Tuple[int, int]], 

148 old_string: str, new_string: str) -> Optional[str]: 

149 """Detect tool-call escape-drift artifacts in new_string. 

150 

151 Looks for ``\\'`` or ``\\"`` sequences that are present in both 

152 old_string and new_string (i.e. the model copy-pasted them as "context" 

153 it intended to preserve) but don't exist in the matched region of the 

154 file. That pattern indicates the transport layer inserted spurious 

155 shell-style escapes around apostrophes or quotes — writing new_string 

156 verbatim would literally insert ``\\'`` into source code. 

157 

158 Returns an error string if drift is detected, None otherwise. 

159 """ 

160 # Cheap pre-check: bail out unless new_string actually contains a 

161 # suspect escape sequence. This keeps the guard free for all the 

162 # common, correct cases.

164 return None 

165 

166 # Aggregate matched regions of the file — that's what new_string will 

167 # replace. If the suspect escapes are present there already, the 

168 # model is genuinely preserving them (valid for some languages / 

169 # escaped strings); accept the patch. 

170 matched_regions = "".join(content[start:end] for start, end in matches) 

171 

172 for suspect in ("\\'", '\\"'): 172 ↛ 184line 172 didn't jump to line 184 because the loop on line 172 didn't complete

174 plain = suspect[1] # "'" or '"' 

175 return ( 

176 f"Escape-drift detected: old_string and new_string contain " 

177 f"the literal sequence {suspect!r} but the matched region of " 

178 f"the file does not. This is almost always a tool-call " 

179 f"serialization artifact where an apostrophe or quote got " 

180 f"prefixed with a spurious backslash. Re-read the file with " 

181 f"read_file and pass old_string/new_string without " 

182 f"backslash-escaping {plain!r} characters." 

183 ) 

184 return None

206def _reindent_replacement(file_region: str, old_string: str, new_string: str) -> str: 

207 """Adjust ``new_string`` so its indentation matches ``file_region``. 

208 

209 Used after a non-exact fuzzy match: the LLM may have sent old_string and 

210 new_string with a different indent than the file actually has (e.g. 

211 2-space indent in tool args vs 4-space indent on disk). The fuzzy 

212 strategy successfully matched anyway, but writing ``new_string`` verbatim 

213 would corrupt the file's indentation. 

214 

215 Approach: 

216 

217 1. For each non-blank line in ``new_string``, compute its indent 

218 *relative* to the shallowest non-blank line of ``old_string`` (the 

219 LLM's base indent). 

220 2. Anchor that relative indent onto the file's actual base indent (the 

221 leading whitespace of the file_region's first non-blank line). 

222 3. Re-emit each non-blank line as ``file_base + (line_indent - llm_base)``. 

223 

224 Blank lines and lines less-indented than the LLM's base are anchored 

225 directly to the file's base indent. 

226 

227 No-op cases (returns ``new_string`` unchanged): 

228 - file_region or old_string has no meaningful line 

229 - LLM base indent equals file base indent 

230 - new_string is empty 

231 """ 

232 if not new_string: 232 ↛ 233line 232 didn't jump to line 233 because the condition on line 232 was never true

233 return new_string 

234 

235 old_first = _first_meaningful_line(old_string) 

236 file_first = _first_meaningful_line(file_region) 

237 if old_first is None or file_first is None: 237 ↛ 238line 237 didn't jump to line 238 because the condition on line 237 was never true

238 return new_string 

239 

240 old_indent = _leading_whitespace(old_first) 

241 file_indent = _leading_whitespace(file_first) 

242 

243 if old_indent == file_indent: 

244 return new_string 

245 

246 # Re-indent each line of new_string. Strategy: replace the LLM's base 

247 # indent prefix with the file's base indent prefix, preserving any 

248 # additional indent the LLM added on top. This is the same approach 

249 # Roo Code uses (multi-search-replace.ts:466-500). It preserves the 

250 # LLM's intended *relative* nesting between lines while anchoring to 

251 # the file's actual indent style. 

252 out_lines: List[str] = [] 

253 for line in new_string.split("\n"): 

254 if not line.strip(): 

255 # Blank lines: leave whitespace untouched. 

256 out_lines.append(line) 

257 continue 

258 line_indent = _leading_whitespace(line) 

259 if line_indent.startswith(old_indent): 259 ↛ 267line 259 didn't jump to line 267 because the condition on line 259 was always true

260 # Common case: line has the LLM's base indent (possibly plus 

261 # extra). Swap base prefix for the file's base prefix. 

262 remainder = line[len(old_indent):] 

263 out_lines.append(file_indent + remainder) 

264 else: 

265 # Line is less-indented than the LLM's base — e.g. a dedent at 

266 # the start of new_string. Anchor to the file's base. 

267 out_lines.append(file_indent + line.lstrip(" \t")) 

268 return "\n".join(out_lines)

271def _maybe_unescape_new_string(new_string: str, 

272 content: str, 

273 matches: List[Tuple[int, int]]) -> str: 

274 """Conditionally unescape ``\\t``/``\\r`` in new_string. 

275 

276 LLMs frequently send the two-character sequences ``\\t`` (backslash + t) 

277 and ``\\r`` (backslash + r) inside JSON tool-call arguments where they 

278 meant a real tab or carriage-return byte. Writing the string verbatim 

279 corrupts tab-indented files with literal backslash-letter pairs. 

280 

281 The unescape is only applied per-sequence when the *matched region of 

282 the file* actually contains the corresponding control character — that 

283 is, we only convert ``\\t`` -> tab when the file region we're replacing 

284 contains a real tab byte. Files that legitimately contain the literal 

285 two-character string ``"\\t"`` (e.g. a Python source line that defines 

286 ``sep = "\\t"``) get a backslash+t in the matched region instead of a 

287 tab, so we leave new_string alone. 

288 

289 ``\\n`` is intentionally excluded: newlines serialize correctly through 

290 JSON and rewriting backslash-n would corrupt escape sequences in 

291 string literals far more often than it would help. 

292 """ 

293 # Cheap pre-check — bail out unless new_string actually contains one of 

294 # the suspect sequences. Keeps the common case free. 

295 if "\\t" not in new_string and "\\r" not in new_string: 

296 return new_string 

297 

298 matched_regions = "".join(content[start:end] for start, end in matches) 

299 out = new_string 

300 if "\\t" in out and "\t" in matched_regions: 

301 out = out.replace("\\t", "\t") 

302 if "\\r" in out and "\r" in matched_regions: 

303 out = out.replace("\\r", "\r") 

304 return out

308 new_string: str, old_string: Optional[str] = None) -> str: 

309 """ 

310 Apply replacements at the given positions. 

311 

312 Args: 

313 content: Original content 

314 matches: List of (start, end) positions to replace 

315 new_string: Replacement text 

316 old_string: When non-None, signals that the match came from a

320 Returns:

322 """

324 # This preserves positions of earlier matches 

325 sorted_matches = sorted(matches, key=lambda x: x[0], reverse=True)

327 result = content 

328 for start, end in sorted_matches: 

329 if old_string is not None:

331 adjusted = _reindent_replacement(file_region, old_string, new_string) 

332 else: 

333 adjusted = new_string

335

524def _strategy_unicode_normalized(content: str, pattern: str) -> List[Tuple[int, int]]: 

525 """Strategy 7: Unicode normalisation. 

526

529 exact and line_trimmed matching on the normalised copies. 

530

533 replacements expand a single character into multiple ASCII characters,

535 """

537 # carry unicode variants — e.g. content has an em-dash that should match 

538 # the LLM's ASCII '--', or vice-versa. Skip only when neither changes. 

539 norm_pattern = _unicode_normalize(pattern)

542 return []

579 for i in range(len(norm_content_lines) - pattern_line_count + 1): 

580 if (norm_content_lines[i].strip() == first_line and 

581 norm_content_lines[i + pattern_line_count - 1].strip() == last_line): 

582 potential_matches.append(i) 

583 

584 matches = [] 

585 candidate_count = len(potential_matches) 

586 

587 # Thresholding logic: 0.50 for unique matches, 0.70 for multiple candidates. 

588 # Previous values (0.10 / 0.30) were dangerously loose — a 10% middle-section 

589 # similarity could match completely unrelated blocks. 

590 threshold = 0.50 if candidate_count == 1 else 0.70 

591 

592 for i in potential_matches: 

593 if pattern_line_count <= 2: 593 ↛ 594line 593 didn't jump to line 594 because the condition on line 593 was never true

594 similarity = 1.0 

595 else: 

596 # Compare normalized middle sections 

597 content_middle = '\n'.join(norm_content_lines[i+1:i+pattern_line_count-1]) 

598 pattern_middle = '\n'.join(pattern_lines[1:-1]) 

599 similarity = SequenceMatcher(None, content_middle, pattern_middle).ratio() 

600 

601 if similarity >= threshold: 

602 # Calculate positions using ORIGINAL lines to ensure correct character offsets in the file 

603 start_pos, end_pos = _calculate_line_positions( 

604 orig_content_lines, i, i + pattern_line_count, len(content) 

605 ) 

606 matches.append((start_pos, end_pos))

611def _strategy_context_aware(content: str, pattern: str) -> List[Tuple[int, int]]: 

612 """ 

613 Strategy 9: Line-by-line similarity with 50% threshold. 

614  

615 Finds blocks where at least 50% of lines have high similarity. 

616 """ 

617 pattern_lines = pattern.split('\n') 

618 content_lines = content.split('\n') 

619 

620 if not pattern_lines: 620 ↛ 621line 620 didn't jump to line 621 because the condition on line 620 was never true

621 return [] 

622 

623 matches = [] 

624 pattern_line_count = len(pattern_lines) 

625 

626 for i in range(len(content_lines) - pattern_line_count + 1): 

627 block_lines = content_lines[i:i + pattern_line_count] 

628 

629 # Calculate line-by-line similarity 

630 high_similarity_count = 0 

631 for p_line, c_line in zip(pattern_lines, block_lines): 

632 sim = SequenceMatcher(None, p_line.strip(), c_line.strip()).ratio() 

633 if sim >= 0.80: 

634 high_similarity_count += 1 

635 

636 # Need at least 50% of lines to have high similarity 

637 if high_similarity_count >= len(pattern_lines) * 0.5: 

638 start_pos, end_pos = _calculate_line_positions( 

639 content_lines, i, i + pattern_line_count, len(content) 

640 ) 

641 matches.append((start_pos, end_pos))

650def _calculate_line_positions(content_lines: List[str], start_line: int, 

651 end_line: int, content_length: int) -> Tuple[int, int]: 

652 """Calculate start and end character positions from line indices. 

653 

654 Args: 

655 content_lines: List of lines (without newlines) 

656 start_line: Starting line index (0-based) 

657 end_line: Ending line index (exclusive, 0-based) 

658 content_length: Total length of the original content string 

659 

660 Returns: 

661 Tuple of (start_pos, end_pos) in the original content 

662 """ 

663 start_pos = sum(len(line) + 1 for line in content_lines[:start_line]) 

664 end_pos = sum(len(line) + 1 for line in content_lines[:end_line]) - 1 

665 end_pos = min(content_length, end_pos) 

666 return start_pos, end_pos

704def _map_normalized_positions(original: str, normalized: str, 

705 normalized_matches: List[Tuple[int, int]]) -> List[Tuple[int, int]]: 

706 """ 

707 Map positions from normalized string back to original. 

708  

709 This is a best-effort mapping that works for whitespace normalization. 

710 """ 

711 if not normalized_matches: 711 ↛ 712line 711 didn't jump to line 712 because the condition on line 711 was never true

712 return [] 

713 

714 # Build character mapping from normalized to original 

715 orig_to_norm = [] # orig_to_norm[i] = position in normalized 

716 

717 orig_idx = 0 

718 norm_idx = 0 

719 

720 while orig_idx < len(original) and norm_idx < len(normalized): 

721 if original[orig_idx] == normalized[norm_idx]: 

722 orig_to_norm.append(norm_idx) 

723 orig_idx += 1 

724 norm_idx += 1 

725 elif original[orig_idx] in ' \t' and normalized[norm_idx] == ' ': 725 ↛ 727line 725 didn't jump to line 727 because the condition on line 725 was never true

726 # Original has space/tab, normalized collapsed to space

728 orig_idx += 1

730 if orig_idx < len(original) and original[orig_idx] not in ' \t': 

731 norm_idx += 1 

732 elif original[orig_idx] in ' \t': 732 ↛ 738line 732 didn't jump to line 738 because the condition on line 732 was always true

733 # Extra whitespace in original 

734 orig_to_norm.append(norm_idx) 

735 orig_idx += 1 

736 else: 

737 # Mismatch - shouldn't happen with our normalization 

738 orig_to_norm.append(norm_idx) 

739 orig_idx += 1 

740 

741 # Fill remaining 

742 while orig_idx < len(original): 742 ↛ 743line 742 didn't jump to line 743 because the condition on line 742 was never true

743 orig_to_norm.append(len(normalized)) 

744 orig_idx += 1 

745 

746 # Reverse mapping: for each normalized position, find original range 

747 norm_to_orig_start = {} 

748 norm_to_orig_end = {} 

749 

750 for orig_pos, norm_pos in enumerate(orig_to_norm): 

751 if norm_pos not in norm_to_orig_start: 

752 norm_to_orig_start[norm_pos] = orig_pos 

753 norm_to_orig_end[norm_pos] = orig_pos

756 original_matches = []

759 if norm_start in norm_to_orig_start: 759 ↛ 763line 759 didn't jump to line 763 because the condition on line 759 was always true

760 orig_start = norm_to_orig_start[norm_start]

764 

765 # Find original end

780def find_closest_lines(old_string: str, content: str, context_lines: int = 2, max_results: int = 3) -> str: 

781 """Find lines in content most similar to old_string for "did you mean?" feedback. 

782 

783 Returns a formatted string showing the closest matching lines with context, 

784 or empty string if no useful match is found. 

785 """ 

786 if not old_string or not content: 

787 return "" 

788 

789 old_lines = old_string.splitlines() 

790 content_lines = content.splitlines() 

791 

792 if not old_lines or not content_lines: 792 ↛ 793line 792 didn't jump to line 793 because the condition on line 792 was never true

793 return "" 

794 

795 # Use first line of old_string as anchor for search 

796 anchor = old_lines[0].strip() 

797 if not anchor: 797 ↛ 799line 797 didn't jump to line 799 because the condition on line 797 was never true

798 # Try second line if first is blank 

799 candidates = [l.strip() for l in old_lines if l.strip()] 

800 if not candidates: 

801 return "" 

802 anchor = candidates[0] 

803 

804 # Score each line in content by similarity to anchor 

805 scored = [] 

806 for i, line in enumerate(content_lines): 

807 stripped = line.strip() 

808 if not stripped: 808 ↛ 809line 808 didn't jump to line 809 because the condition on line 808 was never true

809 continue 

810 ratio = SequenceMatcher(None, anchor, stripped).ratio() 

811 if ratio > 0.3: 

812 scored.append((ratio, i)) 

813 

814 if not scored: 

815 return "" 

816 

817 # Take top matches 

818 scored.sort(key=lambda x: -x[0]) 

819 top = scored[:max_results] 

820 

821 parts = [] 

822 seen_ranges = set() 

823 for _, line_idx in top: 

824 start = max(0, line_idx - context_lines) 

825 end = min(len(content_lines), line_idx + len(old_lines) + context_lines) 

826 key = (start, end) 

827 if key in seen_ranges: 827 ↛ 828line 827 didn't jump to line 828 because the condition on line 827 was never true

828 continue 

829 seen_ranges.add(key) 

830 snippet = "\n".join( 

831 f"{start + j + 1:4d}| {content_lines[start + j]}" 

832 for j in range(end - start) 

833 ) 

834 parts.append(snippet) 

835 

836 if not parts: 836 ↛ 837line 836 didn't jump to line 837 because the condition on line 836 was never true

837 return "" 

838 

839 return "\n---\n".join(parts)
