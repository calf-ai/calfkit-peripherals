96def is_write_denied(path: str) -> bool: 

97 """Return True if path is blocked by the write denylist or safe root.""" 

98 home = os.path.realpath(os.path.expanduser("~")) 

99 resolved = os.path.realpath(os.path.expanduser(str(path))) 

100 

101 if resolved in build_write_denied_paths(home): 

102 return True 

103 for prefix in build_write_denied_prefixes(home): 

104 if resolved.startswith(prefix): 

105 return True 

106 

107 # Hermes control-plane files: block both the ACTIVE profile's view 

108 # (hermes_home) AND the global root view. Without the root pass, a 

109 # profile-mode session leaves <root>/auth.json + <root>/config.yaml 

110 # writable — letting a prompt-injected write_file overwrite the global 

111 # files that every profile inherits from (same shape as #15981). 

112 control_file_names = ("auth.json", "config.yaml", "webhook_subscriptions.json") 

113 mcp_tokens_dir_name = "mcp-tokens" 

114 

115 hermes_dirs = [] 

116 for base in (_hermes_home_path(), _hermes_root_path()): 

117 try: 

118 real = os.path.realpath(base) 

119 if real not in hermes_dirs: 

120 hermes_dirs.append(real) 

121 except Exception: 

122 continue 

123 

124 for base_real in hermes_dirs: 

125 for name in control_file_names: 

126 try: 

127 if resolved == os.path.realpath(os.path.join(base_real, name)): 

128 return True 

129 except Exception: 

130 continue 

131 try: 

132 mcp_real = os.path.realpath(os.path.join(base_real, mcp_tokens_dir_name)) 

133 if resolved == mcp_real or resolved.startswith(mcp_real + os.sep): 

134 return True 

135 except Exception: 

136 pass 

137 try: 

138 pairing_real = os.path.realpath(os.path.join(base_real, "pairing")) 

139 if resolved == pairing_real or resolved.startswith(pairing_real + os.sep): 

140 return True 

141 except Exception:

143 

144 safe_root = get_safe_write_root() 

145 if safe_root and not (resolved == safe_root or resolved.startswith(safe_root + os.sep)):

311# --------------------------------------------------------------------------- 

312# Cross-profile write guard (#TBD) 

313# 

314# Hermes profiles are separate HERMES_HOME dirs under 

315# ``<root>/profiles/<name>/``. Each profile has its own skills/, plugins/, 

316# cron/, memories/. When an agent runs under one profile, writing into 

317# ANOTHER profile's directories is almost always wrong — those skills / 

318# plugins / cron jobs / memories affect a different session the user runs 

319# from a different shell. 

320# 

321# Soft guard, NOT a security boundary: the agent runs as the same OS user 

322# and has unrestricted terminal access, so this returns a warning the model 

323# can choose to honor or override with ``cross_profile=True``. Same shape 

324# as the dangerous-command approval flow — the agent is told the boundary 

325# exists, and explicit user direction is required to cross it. 

326# 

327# Reference: May 2026 incident where a hermes-security profile session 

328# edited skills under both ``~/.hermes/profiles/hermes-security/skills/`` 

329# AND ``~/.hermes/skills/`` (the default profile's skills) without realizing 

330# the second path belonged to a different profile. 

331# --------------------------------------------------------------------------- 

332 

333# Profile-scoped directories under HERMES_HOME / <root> / <root>/profiles/<X>/ 

334# that should be guarded. Adding a new area here extends the guard with no 

335# other code change. 

336PROFILE_SCOPED_AREAS = ("skills", "plugins", "cron", "memories")

340 """Return the active profile name derived from HERMES_HOME.

345 Falls back to ``"default"`` on any resolution failure so the guard

364def classify_cross_profile_target(path: str) -> Optional[dict]: 

365 """Classify a write target as cross-profile if it lands in another 

366 profile's scoped area (skills/plugins/cron/memories). 

367 

368 Returns ``None`` when the target is outside Hermes scope, or is inside 

369 the ACTIVE profile, or doesn't hit a profile-scoped area. Otherwise 

370 returns a dict with: 

371 

372 * ``active_profile``: name of the profile the agent is running as 

373 * ``target_profile``: name of the profile the path belongs to 

374 * ``area``: which scoped area (``"skills"``, ``"plugins"``, etc.) 

375 * ``target_path``: the resolved path string 

376 

377 The caller decides what to do with the result — surface a warning to 

378 the model, prompt the user, or (with explicit consent / 

379 ``cross_profile=True``) proceed anyway. 

380 """ 

381 try: 

382 target = Path(os.path.expanduser(str(path))).resolve() 

383 root_real = _hermes_root_path().resolve() 

384 except (OSError, RuntimeError): 

385 return None 

386 

387 target_profile: Optional[str] = None 

388 area: Optional[str] = None 

389 

390 try: 

391 rel = target.relative_to(root_real) 

392 except ValueError: 

393 return None 

394 

395 parts = rel.parts 

396 if not parts: 

397 return None 

398 

399 if parts[0] in PROFILE_SCOPED_AREAS: 

400 # ``<root>/<area>/...`` → default profile. 

401 target_profile = "default" 

402 area = parts[0] 

403 elif ( 

404 parts[0] == "profiles" 

405 and len(parts) >= 3 

406 and parts[2] in PROFILE_SCOPED_AREAS 

407 ): 

408 # ``<root>/profiles/<name>/<area>/...`` → named profile. 

409 target_profile = parts[1] 

410 area = parts[2] 

411 else: 

412 return None 

413 

414 active_profile = _resolve_active_profile_name() 

415 if target_profile == active_profile: 

416 # In-profile write — not a cross-profile event. 

417 return None 

418 

419 return { 

420 "active_profile": active_profile, 

421 "target_profile": target_profile, 

422 "area": area, 

423 "target_path": str(target), 

424 }

456# --------------------------------------------------------------------------- 

457# Sandbox-mirror write guard (#32049) 

458# 

459# Non-local terminal backends (Docker, Daytona, etc.) bind a sandbox-local 

460# directory to the container's ``$HOME``. The on-disk layout looks like 

461# 

462# <HERMES_HOME>/profiles/<name>/sandboxes/<backend>/<task>/home/.hermes/... 

463# 

464# When the agent (running host-side) speculates that authoritative profile 

465# state lives at one of those sandbox-mirror paths, the write lands on the 

466# mirror — never read by the host process — while the host file is left 

467# untouched. The agent reports success, the user sees no change, and on 

468# disk two divergent copies accumulate. See #32049 for evidence. 

469# 

470# This guard is path-shape-only: it detects the 

471# ``…/sandboxes/<backend>/<task>/home/.hermes/…`` segment and warns 

472# regardless of which Hermes profile is active. It does NOT cover the 

473# inner-container case where the bind mount strips the ``sandboxes/`` prefix 

474# (the agent's view inside the container is plain ``/root/.hermes/...``); 

475# that case needs a separate dispatch-layer or host-side ``profile_state`` 

476# tool. 

477# ---------------------------------------------------------------------------
